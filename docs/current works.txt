ALPHA = 0.1

def load_profile(filepath):
    try:
        with open(filepath, 'r') as f:
            data = json.load(f)
            # Check if file is empty or missing keys
            if not data or 'hold_times' not in data:
                return None
            return data
    except (FileNotFoundError, json.JSONDecodeError):
        return None

def calculate_risk_score(baseline, current):
    # Extract vectors
    base_vec = np.array(list(baseline['hold_times'].values()) + list(baseline['latencies'].values()))
    curr_vec = np.array(list(current['hold_times'].values()) + list(current['latencies'].values()))
    
    # Align vectors
    if len(base_vec) != len(curr_vec):
        max_len = max(len(base_vec), len(curr_vec))
        base_vec = np.pad(base_vec, (0, max_len - len(base_vec)))
        curr_vec = np.pad(curr_vec, (0, max_len - len(curr_vec)))
    
    if len(base_vec) == 0: return 1.0 # High risk if no data
    
    return euclidean(base_vec, curr_vec)

def update_baseline(baseline, current):
    for key in baseline['hold_times']:
        if key in current['hold_times']:
            baseline['hold_times'][key] = (1 - ALPHA) * baseline['hold_times'][key] + ALPHA * current['hold_times'][key]
    return baseline
